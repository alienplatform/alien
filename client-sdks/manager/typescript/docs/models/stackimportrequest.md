# StackImportRequest

Request body for manager-side stack import.

## Example Usage

```typescript
import { StackImportRequest } from "@alienplatform/manager-api/models";

let value: StackImportRequest = {
  deploymentGroupToken: "<value>",
  deploymentName: "<value>",
  managementConfig: {
    managingRoleArn: "<value>",
    platform: "aws",
  },
  platform: "test",
  region: "<value>",
  resources: [],
  stackSettings: {},
};
```
