# StackResourceState

Represents the state of a single resource within the stack for a specific platform.

## Example Usage

```typescript
import { StackResourceState } from "@alienplatform/manager-api/models";

let value: StackResourceState = {
  config: {
    id: "<id>",
    type: "function",
  },
  dependencies: [
    {
      id: "<id>",
      type: "function",
    },
  ],
  error: {
    code: "NOT_FOUND",
    internal: false,
    message: "Item not found.",
  },
  outputs: {
    type: "function",
  },
  previousConfig: {
    id: "<id>",
    type: "function",
  },
  status: "delete-failed",
  type: "<value>",
};
```
