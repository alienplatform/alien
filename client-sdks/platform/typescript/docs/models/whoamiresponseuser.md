# WhoamiResponseUser

Authenticated user subject with workspace-scoped permissions

## Example Usage

```typescript
import { WhoamiResponseUser } from "@alienplatform/platform-api/models";

let value: WhoamiResponseUser = {
  kind: "user",
  id: "<id>",
  email: "Hector93@hotmail.com",
  workspaceId: "<id>",
  role: "workspace.member",
  cliConnected: true,
  onboardingDismissedAt: new Date("2024-04-20T09:35:24.669Z"),
};
```
